## Cost Control Mechanisms

### 1. Rate Limiting
```typescript
const RATE_LIMITS = {
  trial: { rpm: 2, rpd: 2 },      // 2 requests total
  premium: { rpm: 10, rpd: 100 },  // 10/min, 100/day
  pro: { rpm: 30, rpd: 500 },      // 30/min, 500/day
  unlimited: { rpm: 60, rpd: 2000 } // 60/min, 2000/day
};
```

### 2. Credit Deduction Before API Call
```typescript
async function generateWithAI(userId: string, feature: string) {
  const creditCost = CREDIT_COSTS[feature];
  
  // 1. Check & deduct credits FIRST
  const success = await deductCredits(userId, creditCost);
  if (!success) throw new Error('Insufficient credits');
  
  try {
    // 2. Then call API
    const result = await callAI(feature);
    return result;
  } catch (error) {
    // 3. Refund if API fails
    await refundCredits(userId, creditCost);
    throw error;
  }
}
```

### 3. Usage Alerts
```typescript
// Alert thresholds
const ALERTS = {
  warning: 0.8,  // 80% used
  critical: 0.95 // 95% used
};

// Send email when threshold reached
if (usage >= credits * ALERTS.warning) {
  sendEmail('usage_warning', user);
}
```
