# üíæ DATABASE SCHEMA

```sql
-- Provider definitions (managed by superadmin)
CREATE TABLE ai_providers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  slug VARCHAR(50) UNIQUE NOT NULL,
  type VARCHAR(50) NOT NULL, -- "llm", "image", "video", "voice", "music"
  is_enabled BOOLEAN DEFAULT TRUE,
  is_default BOOLEAN DEFAULT FALSE,
  base_url TEXT,
  api_version VARCHAR(50),
  logo_url TEXT,
  description TEXT,
  documentation_url TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Models available per provider
CREATE TABLE ai_models (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  provider_id UUID REFERENCES ai_providers(id) ON DELETE CASCADE,
  model_id VARCHAR(100) NOT NULL,
  name VARCHAR(255) NOT NULL,
  type VARCHAR(50) NOT NULL,
  context_window INTEGER,
  max_output_tokens INTEGER,
  supports_vision BOOLEAN DEFAULT FALSE,
  supports_function_calling BOOLEAN DEFAULT FALSE,
  input_price_per_million DECIMAL(10,4),
  output_price_per_million DECIMAL(10,4),
  price_per_unit DECIMAL(10,4),
  credit_cost INTEGER NOT NULL,
  is_enabled BOOLEAN DEFAULT TRUE,
  is_default BOOLEAN DEFAULT FALSE,
  quality_tier VARCHAR(20),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(provider_id, model_id)
);

-- Platform API keys (managed by superadmin)
CREATE TABLE platform_api_keys (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  provider_id UUID REFERENCES ai_providers(id) ON DELETE CASCADE,
  name VARCHAR(100),
  api_key_encrypted TEXT NOT NULL, -- AES-256 encrypted
  is_active BOOLEAN DEFAULT TRUE,
  is_primary BOOLEAN DEFAULT FALSE,
  last_used_at TIMESTAMP,
  total_requests INTEGER DEFAULT 0,
  total_cost_usd DECIMAL(15,4) DEFAULT 0,
  requests_per_minute INTEGER,
  requests_per_day INTEGER,
  expires_at TIMESTAMP,
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- User API keys (for BYOK - Unlimited plan)
CREATE TABLE user_api_keys (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  provider_id UUID REFERENCES ai_providers(id) ON DELETE CASCADE,
  name VARCHAR(100),
  api_key_encrypted TEXT NOT NULL,
  is_active BOOLEAN DEFAULT TRUE,
  is_valid BOOLEAN DEFAULT TRUE,
  last_validated_at TIMESTAMP,
  validation_error TEXT,
  last_used_at TIMESTAMP,
  total_requests INTEGER DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(organization_id, provider_id)
);
```
